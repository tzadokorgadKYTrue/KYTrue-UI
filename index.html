// ← added: receiver-ui port const KYT_ID_THRESHOLD = Number(process.env.KYT_ID_THRESHOLD) || 750; const REVEALENSE_MODE = process.env.REVEALENSE_MODE || 'mock'; // ── kyt_id generator ────────────────────────────────────────────────────────── // // Format: kyt_id_[RANDOM8]_[CHECKSUM4] // // RANDOM8 — 8 characters drawn from an unambiguous Base32 alphabet // (no 0/O, no 1/I, no L, no 8/B confusions). // Seeded from: crypto.randomBytes(5) → 40 bits of entropy. // // CHECKSUM4 — 4 characters derived deterministically from: // SHA-256( haemo_signature_hash + score + session_id + issued_at_ms ) // truncated to the first 20 bits, then base32-encoded. // This binds the token cryptographically to the biological event // that produced it. Any tampered copy of the token will fail // checksum verification. // // Why this is non-disputable: // • RANDOM8 — collision probability: 1/(32^8) ≈ 1 in 1 trillion per session. // • CHECKSUM4 — changes completely if score, session, or biometric hash changes. // • Together they prove: (a) this specific person was scanned, (b) the scan // produced this specific score, (c) in this specific session. // function generateKytId(params: { haemoHash: string; // haemo_signature_hash from Revealense score: number; // LiveScore (0–1000) sessionId: string; // session_id of this verification issuedAtMs: number; // unix timestamp ms of issuance }): string { const ALPHABET = 'ABCDEFGHJKMNPQRSTUVWXYZ23456789'; // 32 chars, unambiguous // ── RANDOM segment (8 chars from true randomness) ────────────────────────── const rawRandom = randomBytes(5); // 40 bits let randomSeg = ''; let bits = 0n; let bitsCount = 0; for (const byte of rawRandom) { bits = (bits << 8n) | BigInt(byte); bitsCount += 8; while (bitsCount >= 5) { bitsCount -= 5; randomSeg += ALPHABET[Number((bits >> BigInt(bitsCount)) & 31n)]; } } randomSeg = randomSeg.slice(0, 8).padEnd(8, ALPHABET[0]); // ── CHECKSUM segment (4 chars from SHA-256 of biological context) ────────── const payload = [ params.haemoHash, String(params.score), params.sessionId, String(params.issuedAtMs), ].join('|'); const sha = createHash('sha256').update(payload).digest(); // Take first 20 bits of SHA-256 → 4 base32 chars let checksumSeg = ''; let cbits = 0n; let cbitsCount = 0; for (let i = 0; i < 3 && checksumSeg.length < 4; i++) { cbits = (cbits << 8n) | BigInt(sha[i]); cbitsCount += 8; while (cbitsCount >= 5 && checksumSeg.length < 4) { cbitsCount -= 5; checksumSeg += ALPHABET[Number((cbits >> BigInt(cbitsCount)) & 31n)]; } } checksumSeg = checksumSeg.slice(0, 4).padEnd(4, ALPHABET[0]); } // ── Mock Revealense scoring ─────────────────────────────────────────────────── // Returns realistic per-dimension credibility scores (0.0–1.0) and a // fake haemo_signature_hash to simulate what the real Revealense engine returns. function mockRevealenseScore(assessment: Assessment): { haemo: number; behav: number; sincerity: number; haemoHash: string; } { // Different flows produce different score profiles for demo realism const profiles: Record = { transaction_authenticity: { haemo: [0.88, 0.96], behav: [0.84, 0.92], sincerity: [0.86, 0.95] }, kyc_individual: { haemo: [0.90, 0.97], behav: [0.88, 0.96], sincerity: [0.87, 0.96] }, insurance_fnol: { haemo: [0.72, 0.88], behav: [0.60, 0.78], sincerity: [0.30, 0.55] }, default: { haemo: [0.82, 0.95], behav: [0.80, 0.93], sincerity: [0.80, 0.94] }, }; const key = assessment.questionnaire_type.includes('fnol') ? 'insurance_fnol' : assessment.questionnaire_type.includes('kyc') ? 'kyc_individual' : assessment.case_type === 'transaction_authenticity' ? 'transaction_authenticity' : 'default'; const p = profiles[key]; const rnd = (lo: number, hi: number) => lo + Math.random() * (hi - lo); return { haemo: rnd(...p.haemo), behav: rnd(...p.behav), sincerity: rnd(...p.sincerity), // Irreversible cryptographic commitment — represents haemo_signature_hash // from Revealense (in production this comes from the Revealense API response) haemoHash: createHash('sha256') .update(`haemo:${assessment.assessment_id}:${Date.now()}`) .digest('hex'), }; } // Fuse three Revealense dimension scores into LiveScore 0–1000 function fuseToLiveScore(haemo: number, behav: number, sincerity: number): number { // Weighted average: haemodynamic is the strongest signal const weighted = (haemo * 0.45) + (behav * 0.30) + (sincerity * 0.25); return Math.min(1000, Math.max(0, Math.round(weighted * 1000))); } function scoreToDecision(score: number): { decision: Result['decision']; risk_band: Result['risk_band'] } { if (score >= 900) return { decision: 'approved', risk_band: 'Trusted' }; return { decision: 'rejected', risk_band: 'Reject' }; } // ── Routes ──────────────────────────────────────────────────────────────────── // POST /v1/assessments app.post('/v1/assessments', (req, res) => { const { partner_id, case_type, transaction_id, receiver, questionnaire_type, callback_mode, metadata } = req.body;
